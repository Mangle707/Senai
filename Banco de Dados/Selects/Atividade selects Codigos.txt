create database teste;
use teste;
create table cadastro(
id_participante int not null primary key,
nomeCompleto varchar(1000) not null,
Cidade varchar(1000) not null,
idade int not null,
participanteEmClube varchar(100) /*Nessa linha informar se o participante esta ativo ou nao*/
);
-- qual participante realizou a atividade,
-- o valor pago pela atividade,
-- e a data em que ela ocorreu.
create table atividades(
id_atividade int not null primary key,
id_participante int  not null,
valor int not null,
DataOcorrida date not null,
foreign key (id_participante) references cadastro (id_participante)
);
